# 1. Handling Missing Value (Cholesterol = 0)
# Hitung median HANYA dari nilai yang valid (bukan 0)
valid_chol_median <- df %>%
  filter(Cholesterol > 0) %>%
  summarise(MedianChol = median(Cholesterol, na.rm = TRUE)) %>%
  pull(MedianChol)

print(paste("Median kolesterol yang valid:", valid_chol_median))
## [1] "Median kolesterol yang valid: 237"
# Gantikan semua nilai 0 di 'Cholesterol' dengan median yang valid
df <- df %>%
  mutate(Cholesterol = ifelse(Cholesterol == 0, valid_chol_median, Cholesterol))

Building the model

# Pisahkan data latih dan data uji
set.seed(26) # Mirip dengan random_state=26
trainIndex <- createDataPartition(df$HeartDisease, p = .8, 
                                  list = FALSE, 
                                  times = 1)

train_data <- df[ trainIndex,]
test_data  <- df[-trainIndex,]

# Definisikan x_test dan y_test secara terpisah untuk evaluasi nanti
x_test <- test_data %>% select(-HeartDisease)
y_test <- test_data$HeartDisease
# Latih model RandomForest
set.seed(18) # Mirip dengan random_state=18

# Melatih model menggunakan formula (HeartDisease ~ .)
model <- train(HeartDisease ~ ., 
               data = train_data, 
               method = "rf") # rf = RandomForest

print(model)
